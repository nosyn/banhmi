# Documentation Site (apps/docs/) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `apps/docs/` — a TanStack Start documentation site for Banhmi with overview, API reference pages, and Pagefind-based static search.

**Architecture:** TanStack Start (Vite + React, file-based routing under `app/routes/`) provides the shell. Docs content is written as `.mdx` files in `app/docs/`. A `Sidebar` component renders navigation. `@tanstack/react-router` handles routing. Pagefind crawls the built HTML for search. The site is deployed as a static site (`output: 'static'`).

**Tech Stack:** TanStack Start 1.x, React 19, `@tanstack/react-router`, MDX via `@mdx-js/rollup`, Pagefind, Bun runtime

# WebSocket Gateways

## Basic Gateway

```ts
import { WebSocketGateway, SubscribeMessage, WebSocketServer } from 'banhmi'
import type { Server } from 'bun'
import type { WsContext } from 'banhmi'

@WebSocketGateway({ path: '/events' })
export class EventsGateway {
  @WebSocketServer()
  server!: Server

  @SubscribeMessage('message')
  handleMessage(ctx: WsContext) {
    return { event: 'reply', data: `Echo: ${ctx.data}` }
  }
}
```

## Lifecycle Hooks

Implement `OnGatewayInit`, `OnGatewayConnection`, or `OnGatewayDisconnect`:

```ts
import type { OnGatewayConnection, OnGatewayDisconnect } from 'banhmi'

@WebSocketGateway({ path: '/chat' })
export class ChatGateway implements OnGatewayConnection, OnGatewayDisconnect {
  onConnection(ctx: WsContext) {
    console.log('Client connected')
  }
  onDisconnect(ctx: WsContext) {
    console.log('Client disconnected')
  }
}
```

## Message Protocol

All messages use JSON: `{ event: string, data: unknown }`

```json
// Client → Server
{ "event": "message", "data": "hello" }

// Server → Client
{ "event": "message", "data": "echo: hello" }
```

- [ ] **Step 6: Create the catch-all docs route `apps/docs/app/routes/docs/$.tsx`**

```tsx
import { createFileRoute } from '@tanstack/react-router'

export const Route = createFileRoute('/docs/$')({
  component: () => {
    const { _splat } = Route.useParams()
    // Dynamic MDX import based on path
    const DocComponent = useMdxDoc(_splat ?? '')
    return DocComponent ? <DocComponent /> : <div>Page not found</div>
  },
})

// Vite's dynamic import with MDX
function useMdxDoc(slug: string) {
  // Static imports for known pages — expand as docs grow
  const docs: Record<string, React.ComponentType> = {
    'getting-started': require('../docs/getting-started.mdx').default,
    'api/controller': require('../docs/api/controller.mdx').default,
    'api/websockets': require('../docs/api/websockets.mdx').default,
  }
  return docs[slug] ?? null
}
```

Note: For production use, prefer Vite's `import.meta.glob` for automatic MDX discovery. The above implementation is a starting point.

```tsx
import { useEffect, useRef, useState } from 'react'

interface PagefindResult {
  url: string
  meta: { title: string }
  excerpt: string
}

export function Search() {
  const [query, setQuery] = useState('')
  const [results, setResults] = useState<PagefindResult[]>([])
  const pagefind = useRef<{
    search: (q: string) => Promise<{ results: Array<{ data: () => Promise<PagefindResult> }> }>
  } | null>(null)

  useEffect(() => {
    // Pagefind is loaded after build — only available in production
    import('/pagefind/pagefind.js' as unknown as string)
      .then((pf) => { pagefind.current = pf as typeof pagefind.current })
      .catch(() => {}) // not available in dev
  }, [])

  async function handleSearch(q: string) {
    setQuery(q)
    if (!pagefind.current || !q) {
      setResults([])
      return
    }
    const { results: rawResults } = await pagefind.current.search(q)
    const data = await Promise.all(rawResults.slice(0, 5).map((r) => r.data()))
    setResults(data)
  }

  return (
    <div style={{ position: 'relative', marginBottom: '1.5rem' }}>
      <input
        type="search"
        placeholder="Search docs..."
        value={query}
        onChange={(e) => handleSearch(e.target.value)}
        style={{
          width: '100%',
          padding: '0.5rem 0.75rem',
          border: '1px solid #e5e7eb',
          borderRadius: '6px',
          fontSize: '0.9rem',
          boxSizing: 'border-box',
        }}
      />
      {results.length > 0 && (
        <ul
          style={{
            position: 'absolute',
            top: '100%',
            left: 0,
            right: 0,
            background: 'white',
            border: '1px solid #e5e7eb',
            borderRadius: '6px',
            listStyle: 'none',
            margin: 0,
            padding: '0.5rem 0',
            boxShadow: '0 4px 12px rgba(0,0,0,0.1)',
            zIndex: 100,
          }}
        >
          {results.map((r) => (
            <li key={r.url}>
              <a
                href={r.url}
                style={{
                  display: 'block',
                  padding: '0.5rem 1rem',
                  textDecoration: 'none',
                  color: '#111',
                }}
              >
                <div style={{ fontWeight: 600 }}>{r.meta.title}</div>
                <div
                  style={{ fontSize: '0.8rem', color: '#6b7280' }}
                  // Pagefind returns highlighted HTML fragments
                  dangerouslySetInnerHTML={{ __html: r.excerpt }}
                />
              </a>
            </li>
          ))}
        </ul>
      )}
    </div>
  )
}
```
