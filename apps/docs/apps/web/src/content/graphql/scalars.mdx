# Scalars

GraphQL scalars represent leaf values. `@banhmi/graphql` ships the standard
scalars (`String`, `Int`, `Float`, `Boolean`, `ID`) and supports custom scalars.

## Built-in scalars

```ts
import { Field, Float, ID, Int, ObjectType } from '@banhmi/graphql'

@ObjectType()
export class Cat {
  @Field(() => ID)      id: string
  @Field()              name: string    // String is the default
  @Field(() => Int)     age: number
  @Field(() => Float)   weight: number
  @Field()              active: boolean
}
```

## Custom scalar

```ts
import { GraphQLScalarType, Kind } from 'graphql'
import { Scalar } from '@banhmi/graphql'

@Scalar('Date', () => Date)
export class DateScalar {
  description = 'Date custom scalar type'

  parseValue(value: unknown): Date {
    return new Date(value as string)
  }

  serialize(value: unknown): string {
    return (value as Date).toISOString()
  }

  parseLiteral(ast: unknown): Date {
    if ((ast as { kind: string }).kind === Kind.STRING) {
      return new Date((ast as { value: string }).value)
    }
    throw new Error('Invalid date literal')
  }
}
```

Register it as a provider and reference it in field types.

## API reference

- `@Scalar(name, typeFunc)` — register a custom scalar class.
- `parseValue(value)` — deserialise from JSON variable.
- `serialize(value)` — serialise to JSON response.
- `parseLiteral(ast)` — deserialise from inline SDL literal.
