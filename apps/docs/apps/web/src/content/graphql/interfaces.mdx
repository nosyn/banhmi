# Interfaces

GraphQL interfaces define a set of fields that multiple types must implement.

## Usage

```ts
import { Field, ID, InterfaceType, ObjectType } from '@banhmi/graphql'

@InterfaceType()
abstract class Animal {
  @Field(() => ID) id: string
  @Field() name: string
}

@ObjectType({ implements: () => [Animal] })
export class Cat implements Animal {
  @Field(() => ID) id: string
  @Field() name: string
  @Field() indoor: boolean
}

@ObjectType({ implements: () => [Animal] })
export class Dog implements Animal {
  @Field(() => ID) id: string
  @Field() name: string
  @Field() breed: string
}
```

## Resolver for interface

```ts
@Resolver(() => Animal)
export class AnimalsResolver {
  @Query(() => [Animal])
  animals(): Animal[] {
    return [
      Object.assign(new Cat(), { id: '1', name: 'Tom', indoor: true }),
      Object.assign(new Dog(), { id: '2', name: 'Rex', breed: 'Lab' }),
    ]
  }

  @ResolveType()
  resolveType(animal: Cat | Dog) {
    return animal instanceof Cat ? 'Cat' : 'Dog'
  }
}
```

## API reference

- `@InterfaceType()` — declares an abstract interface type.
- `@ObjectType({ implements: () => [Interface] })` — implements an interface.
- `@ResolveType()` — resolves the concrete type for an interface field.
