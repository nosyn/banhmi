# WebSocket Gateways

## Basic Gateway

```ts
import { WebSocketGateway, SubscribeMessage, WebSocketServer } from 'banhmi'
import type { Server } from 'bun'
import type { WsContext } from 'banhmi'

@WebSocketGateway({ path: '/events' })
export class EventsGateway {
  @WebSocketServer()
  server!: Server

  @SubscribeMessage('message')
  handleMessage(ctx: WsContext) {
    return { event: 'reply', data: `Echo: ${ctx.data}` }
  }
}
```

## Lifecycle Hooks

```ts
import type { OnGatewayConnection, OnGatewayDisconnect } from 'banhmi'

@WebSocketGateway({ path: '/chat' })
export class ChatGateway implements OnGatewayConnection, OnGatewayDisconnect {
  onConnection(ctx: WsContext) {
    console.log('Client connected')
  }
  onDisconnect(ctx: WsContext) {
    console.log('Client disconnected')
  }
}
```

## Message Protocol

All messages use JSON: `{ event: string, data: unknown }`
